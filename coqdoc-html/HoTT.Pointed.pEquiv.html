Definition peissect {A B : pType} (f : A <~>* B) : pSect f f^-1*.
Proof.
  pointed_reduce.
  srefine (Build_pHomotopy _ _).
  1: apply (eissect f).
  pointed_reduce.
  unfold moveR_equiv_V.
  apply inverse, concat_1p.
